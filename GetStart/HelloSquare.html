#include #include #include void framebuffer_size_callback(GLFWwindow* window, int width, int height); void processInput(GLFWwindow *window); // 설정 const unsigned int SCR_WIDTH = 800; const unsigned int SCR_HEIGHT = 600; const char *vertexShaderSource = "#version 330 core\n" "layout (location = 0) in vec3 aPos;\n" "void main()\n" "{\n" " gl_Position = vec4(aPos.x, aPos.y, aPos.z, 1.0);\n" "}\0"; const char *fragmentShaderSource = "#version 330 core\n" "out vec4 FragColor;\n" "void main()\n" "{\n" " FragColor = vec4(1.0f, 0.5f, 0.2f, 1.0f);\n" "}\n\0"; int main() { // glfw: 초기화 및 설정 // ------------------------------ glfwInit(); glfwWindowHint(GLFW_CONTEXT_VERSION_MAJOR, 3); glfwWindowHint(GLFW_CONTEXT_VERSION_MINOR, 3); glfwWindowHint(GLFW_OPENGL_PROFILE, GLFW_OPENGL_CORE_PROFILE); #ifdef __APPLE__ glfwWindowHint(GLFW_OPENGL_FORWARD_COMPAT, GL_TRUE); #endif // glfw 창 생성 // -------------------- GLFWwindow* window = glfwCreateWindow(SCR_WIDTH, SCR_HEIGHT, "LearnOpenGL", NULL, NULL); if (window == NULL) { std::cout << "GLFW 창 생성 실패" << std::endl; glfwTerminate(); return -1; } glfwMakeContextCurrent(window); glfwSetFramebufferSizeCallback(window, framebuffer_size_callback); // glad: OpenGL 함수 포인터 로딩 // --------------------------------------- if (!gladLoadGLLoader((GLADloadproc)glfwGetProcAddress)) { std::cout << "GLAD 초기화 실패" << std::endl; return -1; } // 셰이더 프로그램 빌드 및 컴파일 // ------------------------------------ // 정점 셰이더 unsigned int vertexShader = glCreateShader(GL_VERTEX_SHADER); glShaderSource(vertexShader, 1, &vertexShaderSource, NULL); glCompileShader(vertexShader); // 셰이더 컴파일 오류 확인 int success; char infoLog[512]; glGetShaderiv(vertexShader, GL_COMPILE_STATUS, &success); if (!success) { glGetShaderInfoLog(vertexShader, 512, NULL, infoLog); std::cout << "ERROR::SHADER::VERTEX::COMPILATION_FAILED\n" << infoLog << std::endl; } // 프래그먼트 셰이더 unsigned int fragmentShader = glCreateShader(GL_FRAGMENT_SHADER); glShaderSource(fragmentShader, 1, &fragmentShaderSource, NULL); glCompileShader(fragmentShader); // 셰이더 컴파일 오류 확인 glGetShaderiv(fragmentShader, GL_COMPILE_STATUS, &success); if (!success) { glGetShaderInfoLog(fragmentShader, 512, NULL, infoLog); std::cout << "ERROR::SHADER::FRAGMENT::COMPILATION_FAILED\n" << infoLog << std::endl; } // 셰이더 연결 unsigned int shaderProgram = glCreateProgram(); glAttachShader(shaderProgram, vertexShader); glAttachShader(shaderProgram, fragmentShader); glLinkProgram(shaderProgram); // 연결 오류 확인 glGetProgramiv(shaderProgram, GL_LINK_STATUS, &success); if (!success) { glGetProgramInfoLog(shaderProgram, 512, NULL, infoLog); std::cout << "ERROR::SHADER::PROGRAM::LINKING_FAILED\n" << infoLog << std::endl; } glDeleteShader(vertexShader); glDeleteShader(fragmentShader); // 정점 데이터 설정 (버퍼 및 속성 설정) // ------------------------------------------------------------------ float vertices[] = { 0.5f, 0.5f, 0.0f, // 오른쪽 위 0.5f, -0.5f, 0.0f, // 오른쪽 아래 -0.5f, -0.5f, 0.0f, // 왼쪽 아래 -0.5f, 0.5f, 0.0f // 왼쪽 위 }; unsigned int indices[] = { // 0부터 시작하는 인덱스 0, 1, 3, // 첫 번째 삼각형 1, 2, 3 // 두 번째 삼각형 }; unsigned int VBO, VAO, EBO; glGenVertexArrays(1, &VAO); glGenBuffers(1, &VBO); glGenBuffers(1, &EBO); // 먼저 VAO를 바인딩한 후, VBO와 EBO를 바인딩하고 정점 속성을 설정합니다. glBindVertexArray(VAO); glBindBuffer(GL_ARRAY_BUFFER, VBO); glBufferData(GL_ARRAY_BUFFER, sizeof(vertices), vertices, GL_STATIC_DRAW); glBindBuffer(GL_ELEMENT_ARRAY_BUFFER, EBO); glBufferData(GL_ELEMENT_ARRAY_BUFFER, sizeof(indices), indices, GL_STATIC_DRAW); glVertexAttribPointer(0, 3, GL_FLOAT, GL_FALSE, 3 * sizeof(float), (void*)0); glEnableVertexAttribArray(0); // VBO는 바인딩을 해제할 수 있지만, 속성 포인터를 등록한 후 안전하게 바인딩 해제 glBindBuffer(GL_ARRAY_BUFFER, 0); // EBO는 VAO가 활성화된 상태에서 바인딩 해제하지 마세요. VAO에 EBO가 저장됩니다. //glBindBuffer(GL_ELEMENT_ARRAY_BUFFER, 0); // VAO를 바인딩 해제하면 다른 VAO 호출로 인해 실수로 현재 VAO가 수정되지 않도록 할 수 있습니다. // 그러나 VAO나 VBO를 수정하려면 다시 바인딩이 필요하기 때문에 일반적으로 VAO는 직접 필요한 경우를 제외하고는 바인딩 해제하지 않습니다. glBindVertexArray(0); // 선택적으로 와이어프레임 모드로 그리려면 아래 주석을 해제하세요. //glPolygonMode(GL_FRONT_AND_BACK, GL_LINE); // 렌더링 루프 // ----------- while (!glfwWindowShouldClose(window)) { // 입력 처리 // ----- processInput(window); // 렌더링 // ------ glClearColor(0.2f, 0.3f, 0.3f, 1.0f); glClear(GL_COLOR_BUFFER_BIT); // 첫 번째 삼각형 그리기 glUseProgram(shaderProgram); glBindVertexArray(VAO); // VAO가 하나만 있으므로 매번 바인딩할 필요는 없지만, 좀 더 깔끔하게 하기 위해 바인딩합니다. //glDrawArrays(GL_TRIANGLES, 0, 6); glDrawElements(GL_TRIANGLES, 6, GL_UNSIGNED_INT, 0); // glBindVertexArray(0); // 매번 바인딩 해제할 필요는 없습니다. // glfw: 버퍼 교환 및 이벤트 처리 (키보드 입력/마우스 이동 등) // ------------------------------------------------------------------------------- glfwSwapBuffers(window); glfwPollEvents(); } // 선택적으로, 더 이상 사용되지 않는 자원을 해제합니다. // ------------------------------------------------------------------------ glDeleteVertexArrays(1, &VAO); glDeleteBuffers(1, &VBO); glDeleteBuffers(1, &EBO); glDeleteProgram(shaderProgram); // glfw: 종료하고 이전에 할당된 모든 GLFW 자원을 해제합니다. // ------------------------------------------------------------------ glfwTerminate(); return 0; } // 모든 입력을 처리: glfw에서 해당 키가 눌렸는지 확인하고 그에 따라 반응합니다. // --------------------------------------------------------------------------------------------------------- void processInput(GLFWwindow *window) { if (glfwGetKey(window, GLFW_KEY_ESCAPE) == GLFW_PRESS) glfwSetWindowShouldClose(window, true); } // glfw: 윈도우 크기가 변경될 때 (운영체제나 사용자 크기 변경에 의한) 이 콜백 함수가 실행됩니다. // --------------------------------------------------------------------------------------------- void framebuffer_size_callback(GLFWwindow* window, int width, int height) { // 뷰포트가 새 창 크기와 일치하도록 설정합니다. // 이때 width와 height는 레티나 디스플레이에서는 실제로 더 크게 나타날 수 있습니다. glViewport(0, 0, width, height); }
